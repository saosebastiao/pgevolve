```rust
/// Where a dependency edge came from.
///
/// `Structural` edges are derived from the IR shape itself (schemaŌåÉtable,
/// tableŌåÉindex, FK references, sequence ownership). They exist in v0.1.
///
/// `AstExtracted` edges are derived by walking the parsed AST of an object
/// body (view body, function body, expression-index predicate, etc.).
/// First produced in v0.2 view sub-spec.
///
/// `AstDeclared` edges come from explicit `-- @pgevolve dep:` directives
/// that close the PL/pgSQL-dynamic-SQL gap (Decision 11). First produced
/// in v0.2 function sub-spec.
///
/// Ordering: `Structural < AstExtracted < AstDeclared` ŌĆö structural edges
/// are tie-broken first in the Kahn min-heap to preserve v0.1 ordering.
#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord, Hash)]
pub enum DepSource {
    /// Derived from the IR shape; v0.1 default.
    Structural,
    /// Walked out of a parsed body AST.
    AstExtracted,
    /// Declared by a `-- @pgevolve dep:` directive in source SQL.
    AstDeclared,
}

/// An edge in the dependency graph, with provenance metadata.
///
/// Convention matches the existing graph: `from` depends on `to`, so `to`
/// must be created before `from`.
#[derive(Debug, Clone, PartialEq, Eq, PartialOrd, Ord, Hash)]
pub struct DepEdge {
    /// Dependent node (the one that needs `to` to exist first).
    pub from: NodeId,
    /// Dependency target.
    pub to: NodeId,
    /// Provenance of this edge.
    pub source: DepSource,
}
```

```rust
//! AST resolution pass.
//!
//! Runs after the structural parse builds the provisional IR and before
//! `Catalog::canonicalize`. For v0.1 the pass validates that structural
//! references (FKs, default-using sequences) resolve. v0.2 sub-specs
//! extend it to walk body ASTs (view body, function body, etc.) and
//! produce `DepEdge` records with `DepSource::AstExtracted`.

use std::collections::BTreeMap;

use crate::ir::catalog::Catalog;
use crate::parse::statement::SourceLocation;

/// One unresolved reference in the source IR.
#[derive(Debug, Clone, PartialEq, Eq)]
pub struct AstResolutionError {
    /// Free-text description of the missing reference, e.g.
    /// `"FK app.orders.user_id_fkey references app.users which is not declared"`.
    pub message: String,
    /// Source location of the *referencer* (the FK row, the view body line, etc.).
    pub location: Option<SourceLocation>,
}

impl std::fmt::Display for AstResolutionError {
    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
        if let Some(loc) = &self.location {
            write!(f, "{}: {}", loc, self.message)
        } else {
            write!(f, "{}", self.message)
        }
    }
}

/// Run the AST resolution pass over a provisional catalog and its
/// source-location map.
///
/// Returns `Ok(())` when every reference resolves. Returns `Err(Vec<...>)`
/// listing every unresolved reference (we accumulate errors rather than
/// short-circuit so the user sees the full picture in one pass).
pub fn resolve(
    catalog: &Catalog,
    locations: &BTreeMap<String, SourceLocation>,
) -> Result<(), Vec<AstResolutionError>> {
    let mut errors = Vec::new();
    resolve_fk_references(catalog, locations, &mut errors);
    resolve_default_sequence_references(catalog, locations, &mut errors);
    if errors.is_empty() {
        Ok(())
    } else {
        Err(errors)
    }
}

fn resolve_fk_references(
    catalog: &Catalog,
    locations: &BTreeMap<String, SourceLocation>,
    errors: &mut Vec<AstResolutionError>,
) {
    use crate::ir::constraint::ConstraintKind;

    let known_tables: std::collections::BTreeSet<_> =
        catalog.tables.iter().map(|t| t.qname.clone()).collect();

    for table in &catalog.tables {
        for constraint in &table.constraints {
            if let ConstraintKind::ForeignKey(fk) = &constraint.kind {
                if !known_tables.contains(&fk.referenced_table) {
                    let loc_key = format!(
                        "tables.{}.constraints.{}",
                        table.qname, constraint.qname.name,
                    );
                    errors.push(AstResolutionError {
                        message: format!(
                            "FK {} references {} which is not declared in source",
                            constraint.qname.name, fk.referenced_table,
                        ),
                        location: locations.get(&loc_key).cloned(),
                    });
                }
            }
        }
    }
}

fn resolve_default_sequence_references(
    catalog: &Catalog,
    locations: &BTreeMap<String, SourceLocation>,
    errors: &mut Vec<AstResolutionError>,
) {
    use crate::ir::default_expr::DefaultExpr;

    let known_sequences: std::collections::BTreeSet<_> =
        catalog.sequences.iter().map(|s| s.qname.clone()).collect();

    for table in &catalog.tables {
        for column in &table.columns {
            if let Some(DefaultExpr::Sequence(qname)) = &column.default {
                if !known_sequences.contains(qname) {
                    let loc_key = format!(
                        "tables.{}.columns.{}.default",
                        table.qname, column.name,
                    );
                    errors.push(AstResolutionError {
                        message: format!(
                            "column {}.{} defaults to nextval({}) but sequence is not declared",
                            table.qname, column.name, qname,
                        ),
                        location: locations.get(&loc_key).cloned(),
                    });
                }
            }
        }
    }
}
```

```rust
//! Statement-scope body canonicalization.
//!
//! Counterpart to [`NormalizedExpr`](super::normalize_expr::NormalizedExpr).
//! Where `NormalizedExpr` canonicalizes one expression, `NormalizedBody`
//! canonicalizes a statement-shaped body ŌĆö a view's `SELECT`, a function
//! body, an expression-index predicate at full-statement scope.
//!
//! Canonicalization rules (per arch spec Decision 10):
//!
//! - Whitespace collapses; one space between tokens; newlines stripped.
//! - Keywords lowercased.
//! - Type aliases normalized (`int` ŌåÆ `integer`, `int4` ŌåÆ `integer`,
//!   `bool` ŌåÆ `boolean`, etc.) via the existing `ColumnType` machinery.
//! - Redundant parens folded.
//! - Commutative operator operands sorted lexicographically.
//! - Identifiers preserved verbatim (qualification, quoting).
//!
//! For v0.1 this module is unused; v0.2 view/function sub-specs are
//! its first consumers.

use blake3::Hasher;

/// A canonicalized statement-scope body.
#[derive(Debug, Clone, PartialEq, Eq)]
pub struct NormalizedBody {
    canonical_text: String,
    canonical_hash: [u8; 32],
}

/// Error parsing a body.
#[derive(Debug, thiserror::Error)]
pub enum BodyError {
    /// `pg_query` rejected the SQL.
    #[error("pg_query rejected body: {0}")]
    Parse(String),
}

impl NormalizedBody {
    /// Canonicalize a body given its raw SQL text.
    ///
    /// Parses with `pg_query`, walks the AST applying the canonicalization
    /// rules above, then emits a canonical-text string. The hash is
    /// BLAKE3 over the canonical text with a domain separator.
    pub fn from_sql(sql: &str) -> Result<Self, BodyError> {
        let parsed = pg_query::parse(sql).map_err(|e| BodyError::Parse(e.to_string()))?;
        let canonical_text = canonicalize_parse_result(&parsed);
        let canonical_hash = hash_canonical(&canonical_text);
        Ok(Self { canonical_text, canonical_hash })
    }

    /// The canonical text. Two bodies are equivalent iff their canonical
    /// texts are byte-equal.
    pub fn canonical_text(&self) -> &str {
        &self.canonical_text
    }

    /// BLAKE3 hash of the canonical text. Domain-separated.
    pub fn canonical_hash(&self) -> &[u8; 32] {
        &self.canonical_hash
    }
}

fn canonicalize_parse_result(parsed: &pg_query::protobuf::ParseResult) -> String {
    // v0.1-readiness scaffold: round-trip through pg_query's deparser,
    // then apply the universal canonicalization passes that already exist
    // for NormalizedExpr (whitespace, keyword case).
    //
    // V0.2 sub-specs extend this to walk each statement kind explicitly
    // (CreateViewStmt, CreateFunctionStmt, etc.) for full canonicalization.

    let mut out = String::new();
    if let Some(deparsed) = parsed.deparse().ok() {
        out.push_str(&deparsed);
    }
    super::normalize_expr::collapse_whitespace(&out)
}

fn hash_canonical(text: &str) -> [u8; 32] {
    let mut h = Hasher::new();
    h.update(b"pgevolve-normalized-body-v1\n");
    h.update(text.as_bytes());
    *h.finalize().as_bytes()
}
```

```rust
/// `column-position-drift` ŌĆö fires when a table's column order in source
/// disagrees with the target catalog's column order, with no other
/// column-level change accompanying.
///
/// Source is canonical; the lint says "your DB has columns in a different
/// order." Severity is `LintAtPlan` ŌĆö plan refuses unless waived in
/// `intent.toml`.
pub fn column_position_drift_rule(
    source: &Catalog,
    target: &Catalog,
    out: &mut Vec<Finding>,
) {
    use std::collections::BTreeMap;
    let target_tables: BTreeMap<_, _> =
        target.tables.iter().map(|t| (t.qname.clone(), t)).collect();

    for source_table in &source.tables {
        let Some(target_table) = target_tables.get(&source_table.qname) else {
            continue;
        };
        // Names that appear in both, in different positions.
        let source_names: Vec<_> =
            source_table.columns.iter().map(|c| c.name.clone()).collect();
        let target_names: Vec<_> =
            target_table.columns.iter().map(|c| c.name.clone()).collect();
        let common: std::collections::BTreeSet<_> =
            source_names.iter().filter(|n| target_names.contains(n)).cloned().collect();

        let source_order: Vec<_> = source_names.iter().filter(|n| common.contains(n)).collect();
        let target_order: Vec<_> = target_names.iter().filter(|n| common.contains(n)).collect();

        if source_order != target_order {
            out.push(Finding {
                rule_id: "column-position-drift".to_string(),
                severity: Severity::LintAtPlan,
                message: format!(
                    "{}: column position drift. source order [{}] vs catalog order [{}]",
                    source_table.qname,
                    source_order.iter().map(|n| n.as_str()).collect::<Vec<_>>().join(", "),
                    target_order.iter().map(|n| n.as_str()).collect::<Vec<_>>().join(", "),
                ),
                // ... fill any remaining Finding fields per existing pattern
            });
        }
    }
}
```

```rust
//! backends ship in v0.2:

//! - `testcontainers`: boots a fresh container via the testcontainers
//!   crate. Hermetic; requires Docker. Default when Docker is present.

//!   developers without Docker, for projects with extensions that need
```

```rust
//! DSN-backed shadow Postgres.

use async_trait::async_trait;
use tokio_postgres::NoTls;

use crate::config::ShadowConfig;
use super::{PgMajor, ShadowBackend, ShadowGuard};

pub struct DsnBackend {
    base_url: String,
    reset: ResetPolicy,
}

#[derive(Debug, Clone)]
enum ResetPolicy {
    DropSchemaCascade,
    DropDatabaseCreate,
    None,
}

impl DsnBackend {
    pub fn new(config: &ShadowConfig) -> anyhow::Result<Self> {
        let base_url = config
            .url
            .clone()
            .or_else(|| config.url_env.as_ref().and_then(|k| std::env::var(k).ok()))
            .ok_or_else(|| anyhow::anyhow!("[shadow].url or [shadow].url_env required for dsn backend"))?;
        let reset = match config.reset.as_deref().unwrap_or("drop_schema_cascade") {
            "drop_schema_cascade" => ResetPolicy::DropSchemaCascade,
            "drop_database_create" => ResetPolicy::DropDatabaseCreate,
            "none" => ResetPolicy::None,
            other => anyhow::bail!("unknown [shadow].reset: {other}"),
        };
        Ok(Self { base_url, reset })
    }
}

#[async_trait]
impl ShadowBackend for DsnBackend {
    async fn checkout(&self, _major: PgMajor) -> anyhow::Result<Box<dyn ShadowGuard>> {
        // Connect to base_url; verify it's empty (or reset per policy).
        // Return a guard whose Drop runs the reset.
        let guard = DsnGuard {
            url: self.base_url.clone(),
            reset: self.reset.clone(),
        };
        guard.reset_now().await?;
        Ok(Box::new(guard))
    }
}

struct DsnGuard {
    url: String,
    reset: ResetPolicy,
}

impl DsnGuard {
    async fn reset_now(&self) -> anyhow::Result<()> {
        let (client, conn) = tokio_postgres::connect(&self.url, NoTls).await?;
        tokio::spawn(conn);
        match &self.reset {
            ResetPolicy::None => Ok(()),
            ResetPolicy::DropSchemaCascade => {
                // Drop every non-system schema except `public`, then recreate `public`.
                client.batch_execute(
                    "DO $$
                     DECLARE r record;
                     BEGIN
                       FOR r IN SELECT nspname FROM pg_namespace
                                WHERE nspname NOT IN ('pg_catalog', 'information_schema', 'pg_toast', 'public')
                                  AND nspname NOT LIKE 'pg_temp_%' AND nspname NOT LIKE 'pg_toast_temp_%'
                       LOOP EXECUTE format('DROP SCHEMA %I CASCADE', r.nspname); END LOOP;
                       EXECUTE 'DROP SCHEMA public CASCADE';
                       EXECUTE 'CREATE SCHEMA public';
                     END $$;"
                ).await?;
                Ok(())
            }
            ResetPolicy::DropDatabaseCreate => {
                anyhow::bail!("drop_database_create reset policy not implemented for v0.2; use drop_schema_cascade")
            }
        }
    }
}

#[async_trait]
impl ShadowGuard for DsnGuard {
    fn url(&self) -> &str { &self.url }
    async fn reset(&mut self) -> anyhow::Result<()> { self.reset_now().await }
}
```

```rust
//! `pgevolve graph` ŌĆö render the dep graph in DOT or Mermaid format.

use std::path::PathBuf;
use anyhow::Result;
use pgevolve_core::parse::parse_directory;
use pgevolve_core::plan::edges::{build_create_graph, DepEdge, DepSource, NodeId};

use crate::config::Config;
use crate::cli::GraphFormat;

pub fn run(
    config: &Config,
    format: GraphFormat,
    out: Option<PathBuf>,
    plan: Option<PathBuf>,
) -> Result<()> {
    if plan.is_some() {
        anyhow::bail!("--plan rendering not yet implemented (v0.2 sub-spec landing)");
    }

    let catalog = parse_directory(&config.project.schema_dir, &config.managed.ignore_objects)?;
    let graph = build_create_graph(&catalog);
    let edges: Vec<DepEdge> = graph.dep_edges().collect();

    let rendered = match format {
        GraphFormat::Dot => render_dot(&edges),
        GraphFormat::Mermaid => render_mermaid(&edges),
    };

    if let Some(path) = out {
        std::fs::write(&path, &rendered)?;
        eprintln!("wrote {} bytes to {}", rendered.len(), path.display());
    } else {
        print!("{rendered}");
    }
    Ok(())
}

fn render_dot(edges: &[DepEdge]) -> String {
    let mut out = String::from("digraph pgevolve_deps {\n  rankdir=LR;\n  node [shape=box, fontname=Helvetica];\n");
    let mut nodes: std::collections::BTreeSet<String> = Default::default();
    for e in edges {
        nodes.insert(node_label(&e.from));
        nodes.insert(node_label(&e.to));
    }
    for n in &nodes {
        out.push_str(&format!("  \"{n}\";\n"));
    }
    let mut sorted = edges.to_vec();
    sorted.sort();
    for e in sorted {
        let style = match e.source {
            DepSource::Structural => "solid",
            DepSource::AstExtracted => "dashed",
            DepSource::AstDeclared => "dotted",
        };
        out.push_str(&format!(
            "  \"{}\" -> \"{}\" [style={style}];\n",
            node_label(&e.from),
            node_label(&e.to)
        ));
    }
    out.push_str("}\n");
    out
}

fn render_mermaid(edges: &[DepEdge]) -> String {
    let mut out = String::from("graph LR\n");
    let mut sorted = edges.to_vec();
    sorted.sort();
    for e in sorted {
        let arrow = match e.source {
            DepSource::Structural => "-->",
            DepSource::AstExtracted => "-.->",
            DepSource::AstDeclared => "==>",
        };
        out.push_str(&format!(
            "  {} {arrow} {}\n",
            mermaid_safe(&node_label(&e.from)),
            mermaid_safe(&node_label(&e.to))
        ));
    }
    out
}

fn node_label(n: &NodeId) -> String {
    match n {
        NodeId::Schema(s) => format!("schema:{}", s.as_str()),
        NodeId::Table(q) => format!("table:{q}"),
        NodeId::Index(q) => format!("index:{q}"),
        NodeId::Sequence(q) => format!("sequence:{q}"),
        NodeId::Constraint { table, name } => format!("constraint:{table}.{}", name.as_str()),
    }
}

fn mermaid_safe(label: &str) -> String {
    // Mermaid node IDs can't contain `.` or `:` freely; escape into bracketed form.
    let id = label.replace(['.', ':'], "_");
    format!("{id}[\"{label}\"]")
}
```

```rust
//! Cross-check the source IR against a live shadow Postgres.
//!
//! For each Structural edge in the source dep graph, the shadow's
//! `pg_depend` should report a corresponding entry. Missing entries
//! indicate either a dep-graph builder bug or a real source issue
//! the AST didn't catch.

use anyhow::Result;
use pgevolve_core::ir::catalog::Catalog;
use pgevolve_core::plan::edges::{build_create_graph, DepEdge, DepSource};

use crate::shadow::ShadowBackend;

#[derive(Debug, Default)]
pub struct CrossCheckReport {
    pub structural_edges_checked: usize,
    pub missing_from_pg_depend: Vec<String>,
    pub extra_in_pg_depend: Vec<String>,
}

pub async fn cross_check(
    backend: &dyn ShadowBackend,
    source: &Catalog,
    pg_major: u32,
    strict: bool,
) -> Result<CrossCheckReport> {
    let mut guard = backend.checkout(pg_major).await?;
    apply_source_to_shadow(guard.url(), source).await?;

    let (client, conn) = tokio_postgres::connect(guard.url(), tokio_postgres::NoTls).await?;
    tokio::spawn(conn);

    let mut report = CrossCheckReport::default();
    let graph = build_create_graph(source);

    for edge in graph.dep_edges() {
        if !matches!(edge.source, DepSource::Structural) { continue; }
        report.structural_edges_checked += 1;
        let exists = pg_depend_has_edge(&client, &edge).await?;
        if !exists {
            report.missing_from_pg_depend.push(format!("{} -> {}", node_label(&edge.from), node_label(&edge.to)));
        }
    }

    if strict && !report.missing_from_pg_depend.is_empty() {
        anyhow::bail!("--shadow-strict: {} structural edges missing from pg_depend:\n  {}",
            report.missing_from_pg_depend.len(),
            report.missing_from_pg_depend.join("\n  "));
    }
    Ok(report)
}
```
